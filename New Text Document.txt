import java.util.*;

class arrayList
{
	public static void main(String args[])
	{
		List<String> list = new ArrayList<String>();
		list.add("Jabir");
		list.add("Thasni");
		list.add("Hari");
		list.add("Ajay");
		
		iterator itr = list.iterator();
		while(itr.hasNext()){
			System.out.println(itr.next());
		}
	}
}
