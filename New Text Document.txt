import java.util.*;

class NameConcat
{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter integer: ");
		int a;
		a = sc.nextLine();
		System.out.println("Enter float: ");
		float b = sc.nextLine();

		System.out.println("Integer: " + a + "\nFloat: " + b );
	}
}
